# テーマ
レイアウトに関して

## ソースコード例
```js
// App.jsx
import { BrowserRouter, Routes, Route, Link, Outlet } from "react-router-dom";

/* ---------------------------------------------------------
   【Layout】
   - 全ページ共通のレイアウト
   - ナビゲーションなど固定UIを配置
   - <Outlet /> に子ルートが表示される
--------------------------------------------------------- */
function Layout() {
  return (
    <div>
      <nav style={{ display: "flex", gap: 12 }}>
        <Link to="/">Home</Link>
        <Link to="/dashboard">Dashboard</Link>
        <Link to="/dashboard/settings">Settings</Link>
      </nav>
      <hr />

      {/* 子ルートがここに描画される */}
      <Outlet /> 
    </div>
  );
}

/* ----- 各ページコンポーネント ----- */
function Home() {
  return <h2>Home</h2>;
}

function Dashboard() {
  return (
    <div>
      <h2>Dashboard</h2>
      {/* Dashboard配下のルートはこのOutletに表示 */}
      <Outlet />
    </div>
  );
}

function DashboardIndex() {
  return <p>Dashboardのトップページ</p>;
}

function Settings() {
  return <p>Dashboardの設定画面</p>;
}

/* ---------------------------------------------------------
   ルーティング設定
   Layout配下に子ルートを定義することで共通UIを保つ
--------------------------------------------------------- */
export default function App() {
  return (
    <BrowserRouter>
      <Routes>
        {/* 親ルート：共通レイアウト */}
        <Route element={<Layout />}>
          <Route path="/" element={<Home />} />

          {/* /dashboard配下のルート */}
          <Route path="dashboard" element={<Dashboard />}>
            <Route index element={<DashboardIndex />} />      {/* /dashboard */}
            <Route path="settings" element={<Settings />} />  {/* /dashboard/settings */}
          </Route>
        </Route>
      </Routes>
    </BrowserRouter>
  );
}
```

### ライブラリの意味
- BrowserRouter：アプリ全体のルーティング機能を提供。URLを監視して画面を切り替える土台
- Routes：複数のRouteをまとめるコンポーネント。ルート定義の親
- Route：URLパスと表示するコンポーネントを紐づける要素
- Link：画面遷移用タグ。SPAのためページリロードなしで移動できる
- Outlet：ネストされた子ルートの表示位置。Layout内で子ページがここに挿入される
